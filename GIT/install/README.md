## GIT INSTALL , INITIALIZATION , USER DETAILS ON LINUX
======================================================

sudo su

yum install git -y 

git --version

mkdir mygit-linux

cd mygit-linux

git init


git config --global user.name "Your Name"

git config --global user.email "your.email@example.com"

git config --list
